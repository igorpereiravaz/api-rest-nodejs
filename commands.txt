- create package.json
npm init -y

- execute binaries files and create tsconfig file
npx tsc --init

- install fastify
npm i fastify

- install package development dependency for node modules (development ambient)
npm install -D @types/node

- tool for convert code from node (development ambient)
npm install tsx -D

- install eslint and tool default rocketseat
npm i eslint @rocketseat/eslint-config -D

- install extension
npm i dotenv

- install zod for manipulation enviroments config
npm i zod
